\begin{algorithm}[!htbp]
    \caption{Алгоритм $\AlgoExtendedMin(\SetPositiveScenarios, P)$}%
    \label{alg:extended-min}

    \DontPrintSemicolon
    \SetKwFunction{FnGetC}{getNumStates}
    \SetKwFunction{FnGetN}{getNumConditions}

    \Input{$\SetPositiveScenarios$ \--- набор позитивных сценариев выполнения, $P$ \--- максимальный размер охранного условия.}
    \Output{$\Automaton$ \--- конечный автомат с минимальным числом состояний~$\Cmin$ и суммарным размером охранных условий~$\Nmin$.}
    \BlankLine

    $\Automaton' \gets \AlgoBasicMin(\SetPositiveScenarios)$ \;
    $\Cmin \gets \FnGetC(\Automaton')$ \;
    $\Automaton \gets \AlgoExtendedMinN(\SetPositiveScenarios, \Cmin, P)$ \;
    \Return $\Automaton$ \;

    \BlankLine
    \tcc{Функция $\AlgoExtendedMinN(\SetPositiveScenarios, C, P)$ аналогична $\AlgoBasicMinT$, только вместо~$T$ используется~$N$.}
\end{algorithm}
